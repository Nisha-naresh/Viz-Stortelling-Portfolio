<main class="content">

  <p class="drop-cap">Before bar charts became ordinary and before dashboards became wallpaper, there was one image that showed what data could do when it stopped performing and started mourning. Charles Joseph Minard drew it in 1869, at eighty-eight years old. He called it a figurative chart. Historians would later describe it as something closer to an epitaph.</p>

  <div class="pull-quote">
    <p>"The best statistical graphic ever drawn."</p>
    <cite>Edward Tufte, The Visual Display of Quantitative Information, 1983</cite>
  </div>

  <hr class="rule">
  <h2 class="s-label">The Man</h2>
  <h2>Charles Joseph Minard (1781 to 1870)</h2>

    </div>
    <div class="port-body">
      <p>Born in Dijon, Minard spent his career as a civil engineer of bridges and roads, rising to the highest rank in the national inspectorate. He retired in 1851 and, in retirement, produced over fifty figurative maps encoding flows of trade, migration, and war.</p>
      <p>He drew the Napoleon campaign map at eighty-eight years old, two years before his death. It remains the benchmark against which every data visualization is measured.</p>

    <h2 style="margin-top:6px">How the Interactive Map Works</h2>
    <p style="font-size:.97rem;line-height:1.78;color:rgba(26,18,8,.75)" class="reveal">This redesign preserves Minard's core visual argument: the width of the ribbon tells the story of human loss. Every segment of the advance and retreat is interactive. Hover over any part of the map to see exactly how many soldiers remained at that stage, the date, and the percentage of the original army who survived that far.</p>
    <p style="font-size:.97rem;line-height:1.78;color:rgba(26,18,8,.75)" class="reveal">The temperature chart beneath the map shares the same geographic axis, so the relationship between falling temperatures and the collapsing army is visible at a glance. Temperatures originally in Reaumur degrees have been converted to Celsius for a modern reader. Every recorded temperature during the retreat fell below zero.</p>
    <p style="font-size:.97rem;line-height:1.78;color:rgba(26,18,8,.75)" class="reveal">The most important design choice was keeping Minard's proportional width as the primary encoding. A number says 55,000. A ribbon that has almost disappeared says it with grief.</p>

  const tip = document.getElementById('map-tooltip');

  function showTip(e, el){
    const isAdv = el.classList.contains('adv');

    const pct   = parseInt(el.dataset.p);
    tip.innerHTML =
      '<strong style="color:'+col+'">'+el.dataset.n+'</strong>'+

    tip.classList.add('vis');
    moveTip(e);
  }

  function moveTip(e){
    const tw=230, th=110;
    const l = e.clientX+16+tw > window.innerWidth  ? e.clientX-tw-8 : e.clientX+16;
    const t = e.clientY+10+th > window.innerHeight ? e.clientY-th-8 : e.clientY+10;

      el.style.transition = 'fill .15s ease';
      showTip(e, el);
    });
    el.addEventListener('mousemove', moveTip);
    el.addEventListener('mouseleave', () => {
      el.style.fill = 'transparent';
      tip.classList.remove('vis');

  function fireRipple(dotEl){
    const cx = dotEl.getAttribute('cx');
    const cy = dotEl.getAttribute('cy');
    const svg = dotEl.closest('svg');

    // Create two ripple rings
    [0, 200].forEach((delay, i) => {
      setTimeout(() => {
        const c = document.createElementNS('http://www.w3.org/2000/svg','circle');
        c.setAttribute('cx', cx);
        c.setAttribute('cy', cy);
        c.setAttribute('r',  dotEl.getAttribute('r'));
        c.setAttribute('fill','none');
        c.setAttribute('stroke','#b5221e');
        c.setAttribute('stroke-width', i===0 ? '2' : '1.2');
        c.style.opacity = '0';
        svg.appendChild(c);

        // Animate expansion
        const startTime = performance.now();
        const startR = parseFloat(dotEl.getAttribute('r'));
        const endR   = startR + (i===0 ? 18 : 28);
        function step(now){
          const p = Math.min((now - startTime) / 700, 1);
          const eased = 1 - Math.pow(1-p, 2);
          c.setAttribute('r', startR + (endR - startR) * eased);
          c.style.opacity = (1 - p) * (i===0 ? 0.7 : 0.45);
          if(p < 1) requestAnimationFrame(step);
          else c.remove();
        }
        requestAnimationFrame(step);
      }, delay);
    });
  }

  function openPanel(dot){
    const city     = dot.dataset.city;
    const info     = dot.dataset.info;
    const soldiers = dot.dataset.soldiers;
    const pct      = parseInt(dot.dataset.pct);

    panelCity.textContent = city;
    panelInfo.textContent = info;
    panelNum.textContent  = soldiers;
    panelLbl.textContent  = pct + '% of original army';

    // Animate bar
    panelFill.style.width = '0%';
    panel.style.display   = 'block';
    // Force reflow then animate bar
    void panel.offsetWidth;
    setTimeout(() => { panelFill.style.width = pct + '%'; }, 40);

    // Scroll to panel
    setTimeout(() => {
      panel.scrollIntoView({ behavior:'smooth', block:'nearest' });
    }, 100);

    // Fire ripple on dot
    fireRipple(dot);
  }

  function closePanel(){
    panel.style.display = 'none';
  }

  });

      d.setAttribute('fill', d.id==='dot-berezina' ? '#8e1c1c' : '#1a1208');

  });

    if (!document.getElementById('interactive-map').contains(e.target)) {
      closePanel();
      document.querySelectorAll('.city-dot').forEach(d => {
        d.setAttribute('fill', d.id==='dot-berezina' ? '#8e1c1c' : '#1a1208');

  /* ---- SCROLL REVEAL ---- */
  const ro = new IntersectionObserver(entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add('on');
        ro.unobserve(entry.target);
      }
    });
  }, { threshold: 0.1 });
  document.querySelectorAll('.reveal').forEach(el => ro.observe(el));

  /* ---- MAP FADE IN ---- */
  const sec = document.getElementById('interactive-map');
  sec.style.cssText += 'opacity:0;transition:opacity .8s ease';
